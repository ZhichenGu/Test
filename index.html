#include #include #include // --- 用户配置 --- const char* ssid = "iNet"; const char* password = "goodlife"; // 与网页端保持一致的 MQTT 配置 const char* mqtt_server = "08e06f289404444fab8874648aed4d71.s1.eu.hivemq.cloud"; const int mqtt_port = 8883; // 硬件端使用加密端口 8883 const char* mqtt_user = "michael"; const char* mqtt_pass = "12345678Aa_"; const char* target_topic = "button/state"; // 必须与网页中的 TOPIC 一致 const int outputPin = 16; // GPIO16 (NodeMCU 上的 D0) WiFiClientSecure espClient; PubSubClient client(espClient); void setup_wifi() { Serial.print("\nConnecting to "); Serial.println(ssid); WiFi.begin(ssid, password); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println("\nWiFi connected"); } // 核心逻辑：处理从网页发来的消息 void callback(char* topic, byte* payload, unsigned int length) { // 将 payload 转为字符 char message = (char)payload[0]; Serial.print("Message arrived ["); Serial.print(topic); Serial.print("] Value: "); Serial.println(message); if (message == '1') { digitalWrite(outputPin, HIGH); // 网页端按下 Serial.println("GPIO16 -> HIGH"); } else if (message == '0') { digitalWrite(outputPin, LOW); // 网页端松开 Serial.println("GPIO16 -> LOW"); } } void reconnect() { while (!client.connected()) { Serial.print("Attempting MQTT connection..."); String clientId = "ESP8266-Client-" + String(random(0xffff), HEX); if (client.connect(clientId.c_str(), mqtt_user, mqtt_pass)) { Serial.println("connected"); client.subscribe(target_topic); // 订阅网页发送的主题 } else { Serial.print("failed, rc="); Serial.print(client.state()); Serial.println(" try again in 5 seconds"); delay(5000); } } } void setup() { Serial.begin(115200); pinMode(outputPin, OUTPUT); digitalWrite(outputPin, LOW); // 初始设为低电平 // 由于使用 HiveMQ 云端，需要忽略证书指纹校验（简单安全） espClient.setInsecure(); setup_wifi(); client.setServer(mqtt_server, mqtt_port); client.setCallback(callback); } void loop() { if (!client.connected()) { reconnect(); } client.loop(); }
